##字符串转换成变量名##

eval:把字符串当成命令来执行

NameSource=[10 20 30];
for i=1:3
    Name=['VarName',num2str(i)];
    eval([Name,'=NameSource(i)'])
end

VarName1=10;VarName2=20;VarName3=30;

加上小括号

a=['www',num2str(1)];
b.(a)=10;

b=
www1=10

##变量名转换为字符串##

who 返回变量名

a1=123;a2=223;a3=11;
b=who('a*')

b=
    'a1'
    'a2'
    'a3'
